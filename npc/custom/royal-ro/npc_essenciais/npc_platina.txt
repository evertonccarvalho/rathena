prontera,150,195,5	script	Habilidades de Platina	10185,{

	mes "[Habilidades de Platina]";
	mes "Posso lhe dar as habilidades especiais disponíveis para o seu trabalho.";
	mes "Você gostaria dessas habilidades agora?";
	next;
	if (select("Sim:N o") == 2) {
		mes "[Habilidades de Platina]";
		mes "Tenha um bom dia... >.>";
		close;
	}
	callfunc "F_GetPlatinumSkills";
	mes "[Habilidades de Platina]";
	mes "Ai está!";
	close;

OnInit:
	.NPCName$ = callfunc("F_NpcInitStatus", "Velha Negociante", 1);
end;

}
